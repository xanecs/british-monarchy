Male Primogeniture

function MalePrimogeniture (monarch) {
  if (!monarch.hasChildren) {
    return MalePrimogeniture(monarch.previous);
  }

  if (monarch.hasSons) {
    return monarch.sons.eldest;
  }
  else if (monarch.hasDaughters) {
    return monarch.daughters.eldest;
  }
}
